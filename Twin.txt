What Can Go Wrong in Clusters
Before the prompts, understand the four failure modes you’re validating against:

1. OVER-MERGING    → Two distinct topics forced into one cluster
                     e.g. "KYC Requirements" + "Monetary Policy" in same cluster
                     
2. UNDER-MERGING   → Same topic split across two clusters
                     e.g. Cluster 12: "Repo Rate Policy"
                          Cluster 31: "RBI Policy Rate Decisions"
                          
3. NOISE POLLUTION → One outlier topic dragging down cluster coherence
                     e.g. 8 KYC topics + 1 unrelated "NBFC Registration" topic
                     
4. GENERIC CLUSTER → Topics too vague to be useful for RAG
                     e.g. "Banking Regulations", "RBI Guidelines", "Compliance"

full validation pipeline

import json
import re
import time
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity
from itertools import combinations

class ClusterValidator:
    
    def __init__(self, model_endpoint, similarity_threshold=0.75,
                 delay=0.3):
        self.endpoint = model_endpoint
        self.similarity_threshold = similarity_threshold
        self.delay = delay
    
    # --------------------------------------------------------
    # Core API Call
    # --------------------------------------------------------
    
    def call_llm(self, prompt, temperature=0.1, max_tokens=400):
        import requests
        response = requests.post(
            self.endpoint,
            json={
                "prompt": prompt,
                "temperature": temperature,
                "max_tokens": max_tokens
            },
            timeout=45
        )
        return response.json()['text'].strip()
    
    def parse_json_response(self, text):
        text = re.sub(r'```json|```', '', text).strip()
        try:
            return json.loads(text)
        except:
            # Try to extract JSON object
            match = re.search(r'\{.*\}', text, re.DOTALL)
            if match:
                return json.loads(match.group())
            raise ValueError(f"Could not parse JSON from: {text[:200]}")
    
    # --------------------------------------------------------
    # Step 1: Single Cluster Coherence Validation
    # --------------------------------------------------------
    
    def validate_single_cluster(self, cluster_id, canonical, topics):
        
        topics_formatted = "\n".join([
            f"{i+1}. [{t.get('name', t)}]: {t.get('description', '')}"
            for i, t in enumerate(topics[:20])
        ])
        
        prompt = CLUSTER_COHERENCE_PROMPT.format(
            canonical_name=canonical['canonical_name'],
            canonical_description=canonical['canonical_description'],
            cluster_size=len(topics),
            topics_formatted=topics_formatted
        )
        
        try:
            response = self.call_llm(prompt)
            result = self.parse_json_response(response)
            result['cluster_id'] = cluster_id
            result['validated'] = True
            return result
            
        except Exception as e:
            return {
                'cluster_id': cluster_id,
                'verdict': 'VALIDATION_FAILED',
                'confidence': 0.0,
                'reasoning': str(e),
                'problematic_topics': [],
                'suggested_split_into': [],
                'topics_to_remove': [],
                'validated': False
            }
    
    def run_coherence_validation(self, canonical_topics):
        """
        Run single-cluster validation on all clusters.
        Prioritizes large clusters and low-confidence ones first.
        """
        
        print("=" * 50)
        print("STEP 1: Single Cluster Coherence Validation")
        print("=" * 50)
        
        # Sort by priority: large clusters + low confidence first
        sorted_clusters = sorted(
            canonical_topics.items(),
            key=lambda x: (
                -x[1]['cluster_size'],
                x[1].get('confidence', 1.0)
            )
        )
        
        coherence_results = {}
        verdict_counts = {
            'GOOD': 0, 'OVER_MERGED': 0,
            'NOISE_TOPIC': 0, 'TOO_GENERIC': 0,
            'VALIDATION_FAILED': 0
        }
        
        for cluster_id, canonical in sorted_clusters:
            topics = canonical['source_topics']
            
            # Skip singletons - nothing to validate
            if len(topics) == 1:
                coherence_results[cluster_id] = {
                    'cluster_id': cluster_id,
                    'verdict': 'GOOD',
                    'confidence': 1.0,
                    'reasoning': 'Singleton cluster, auto-approved',
                    'problematic_topics': [],
                    'suggested_split_into': [],
                    'topics_to_remove': [],
                    'skipped': True
                }
                verdict_counts['GOOD'] += 1
                continue
            
            result = self.validate_single_cluster(
                cluster_id, canonical, topics
            )
            coherence_results[cluster_id] = result
            verdict_counts[result['verdict']] = \
                verdict_counts.get(result['verdict'], 0) + 1
            
            status_emoji = {
                'GOOD': '✅',
                'OVER_MERGED': '🔴',
                'NOISE_TOPIC': '🟡',
                'TOO_GENERIC': '🟠',
                'VALIDATION_FAILED': '❌'
            }.get(result['verdict'], '❓')
            
            print(f"{status_emoji} Cluster {cluster_id} "
                  f"[{canonical['canonical_name']}] "
                  f"→ {result['verdict']} "
                  f"(conf: {result['confidence']:.2f})")
            
            if result['verdict'] != 'GOOD':
                print(f"   Reason: {result['reasoning']}")
            
            time.sleep(self.delay)
        
        print(f"\nCoherence Summary: {verdict_counts}")
        return coherence_results
    
    # --------------------------------------------------------
    # Step 2: Cross-Cluster Similarity (Under-Merge Detection)
    # --------------------------------------------------------
    
    def find_similar_cluster_pairs(self, canonical_topics, 
                                    cluster_embeddings):
        """
        Use embedding similarity to find candidate pairs
        for under-merge, then validate with LLM.
        Avoids O(n^2) LLM calls by pre-filtering with embeddings.
        """
        
        print("\n" + "=" * 50)
        print("STEP 2: Cross-Cluster Similarity Check")
        print("=" * 50)
        
        # Compute centroid for each cluster
        cluster_ids = list(canonical_topics.keys())
        centroids = np.array([
            cluster_embeddings[cid].mean(axis=0)
            for cid in cluster_ids
        ])
        
        # Pairwise centroid similarity
        sim_matrix = cosine_similarity(centroids)
        
        # Find pairs above threshold (excluding diagonal)
        candidate_pairs = []
        for i, j in combinations(range(len(cluster_ids)), 2):
            sim = sim_matrix[i][j]
            if sim >= self.similarity_threshold:
                candidate_pairs.append((
                    cluster_ids[i],
                    cluster_ids[j],
                    sim
                ))
        
        # Sort by similarity descending
        candidate_pairs.sort(key=lambda x: x[2], reverse=True)
        
        print(f"Found {len(candidate_pairs)} candidate pairs "
              f"above similarity threshold {self.similarity_threshold}")
        
        # LLM validation for each candidate pair
        cross_cluster_results = []
        
        for cid_a, cid_b, sim in candidate_pairs:
            can_a = canonical_topics[cid_a]
            can_b = canonical_topics[cid_b]
            
            # Get 3 example topics from each
            examples_a = [t.get('name', t) for t in 
                         can_a['source_topics'][:3]]
            examples_b = [t.get('name', t) for t in 
                         can_b['source_topics'][:3]]
            
            prompt = CROSS_CLUSTER_SIMILARITY_PROMPT.format(
                name_a=can_a['canonical_name'],
                desc_a=can_a['canonical_description'],
                examples_a=", ".join(examples_a),
                name_b=can_b['canonical_name'],
                desc_b=can_b['canonical_description'],
                examples_b=", ".join(examples_b)
            )
            
            try:
                response = self.call_llm(prompt)
                result = self.parse_json_response(response)
                result.update({
                    'cluster_a': cid_a,
                    'cluster_b': cid_b,
                    'embedding_similarity': round(sim, 4),
                    'name_a': can_a['canonical_name'],
                    'name_b': can_b['canonical_name']
                })
                cross_cluster_results.append(result)
                
                emoji = '🔗' if result['verdict'] == 'MERGE' else '↔️'
                print(f"{emoji} [{can_a['canonical_name']}] vs "
                      f"[{can_b['canonical_name']}] "
                      f"(sim={sim:.3f}) → {result['verdict']}")
                
                if result['verdict'] == 'MERGE':
                    print(f"   Proposed: {result.get('merged_canonical_name')}")
                    
            except Exception as e:
                print(f"❌ Validation failed for pair "
                      f"({cid_a}, {cid_b}): {e}")
            
            time.sleep(self.delay)
        
        merge_count = sum(
            1 for r in cross_cluster_results 
            if r['verdict'] == 'MERGE'
        )
        print(f"\nCross-cluster summary: "
              f"{merge_count}/{len(candidate_pairs)} pairs recommended for merge")
        
        return cross_cluster_results
    
    # --------------------------------------------------------
    # Step 3: Full Batch Audit
    # --------------------------------------------------------
    
    def run_batch_audit(self, canonical_topics, domain="RBI regulatory"):
        
        print("\n" + "=" * 50)
        print("STEP 3: Full Taxonomy Batch Audit")
        print("=" * 50)
        
        all_names = "\n".join([
            f"{i+1}. {ct['canonical_name']}"
            for i, ct in enumerate(canonical_topics.values())
        ])
        
        prompt = BATCH_AUDIT_PROMPT.format(
            n_topics=len(canonical_topics),
            all_topic_names=all_names,
            domain=domain
        )
        
        try:
            response = self.call_llm(
                prompt, temperature=0.1, max_tokens=1000
            )
            result = self.parse_json_response(response)
            
            print(f"Overall taxonomy quality: "
                  f"{result.get('overall_taxonomy_quality')}")
            print(f"Reasoning: {result.get('overall_reasoning')}")
            print(f"Potential duplicates: "
                  f"{len(result.get('potential_duplicates', []))}")
            print(f"Hierarchy violations: "
                  f"{len(result.get('hierarchy_violations', []))}")
            print(f"Missing coverage areas: "
                  f"{len(result.get('missing_coverage', []))}")
            print(f"Over-granular topics: "
                  f"{len(result.get('over_granular_topics', []))}")
            
            return result
            
        except Exception as e:
            print(f"Batch audit failed: {e}")
            return None
    
    # --------------------------------------------------------
    # Step 4: Generate Action Plan
    # --------------------------------------------------------
    
    def generate_action_plan(self, coherence_results,
                             cross_cluster_results,
                             batch_audit_result):
        """
        Consolidate all validation results into 
        a clear action plan.
        """
        
        actions = {
            'split': [],
            'remove_noise': [],
            'merge': [],
            'review_generic': [],
            'duplicates_from_audit': [],
            'missing_topics': []
        }
        
        # From coherence validation
        for cid, result in coherence_results.items():
            
            if result['verdict'] == 'OVER_MERGED':
                actions['split'].append({
                    'cluster_id': cid,
                    'reason': result['reasoning'],
                    'split_into': result.get('suggested_split_into', []),
                    'confidence': result['confidence']
                })
            
            elif result['verdict'] == 'NOISE_TOPIC':
                actions['remove_noise'].append({
                    'cluster_id': cid,
                    'topics_to_remove': result.get('topics_to_remove', []),
                    'reason': result['reasoning'],
                    'confidence': result['confidence']
                })
            
            elif result['verdict'] == 'TOO_GENERIC':
                actions['review_generic'].append({
                    'cluster_id': cid,
                    'reason': result['reasoning'],
                    'confidence': result['confidence']
                })
        
        # From cross-cluster validation
        for result in cross_cluster_results:
            if result['verdict'] == 'MERGE':
                actions['merge'].append({
                    'cluster_a': result['cluster_a'],
                    'cluster_b': result['cluster_b'],
                    'name_a': result['name_a'],
                    'name_b': result['name_b'],
                    'proposed_name': result.get('merged_canonical_name'),
                    'proposed_desc': result.get('merged_canonical_description'),
                    'similarity': result['embedding_similarity'],
                    'confidence': result['confidence']
                })
        
        # From batch audit
        if batch_audit_result:
            actions['duplicates_from_audit'] = \
                batch_audit_result.get('potential_duplicates', [])
            actions['missing_topics'] = \
                batch_audit_result.get('missing_coverage', [])
        
        # Print action plan
        print("\n" + "=" * 50)
        print("ACTION PLAN")
        print("=" * 50)
        
        print(f"\n🔴 SPLIT ({len(actions['split'])} clusters):")
        for a in actions['split']:
            print(f"   Cluster {a['cluster_id']} → "
                  f"{a['split_into']}")
        
        print(f"\n🟡 REMOVE NOISE ({len(actions['remove_noise'])} clusters):")
        for a in actions['remove_noise']:
            print(f"   Cluster {a['cluster_id']} → "
                  f"remove {a['topics_to_remove']}")
        
        print(f"\n🔗 MERGE ({len(actions['merge'])} pairs):")
        for a in actions['merge']:
            print(f"   [{a['name_a']}] + [{a['name_b']}]"
                  f" → [{a['proposed_name']}]")
        
        print(f"\n🟠 REVIEW GENERIC ({len(actions['review_generic'])} clusters):")
        for a in actions['review_generic']:
            print(f"   Cluster {a['cluster_id']}")
        
        print(f"\n📋 MISSING TOPICS ({len(actions['missing_topics'])}):")
        for m in actions['missing_topics']:
            print(f"   - {m.get('missing_topic')}: {m.get('reason')}")
        
        return actions
    
    # --------------------------------------------------------
    # Master Run Function
    # --------------------------------------------------------
    
    def run_full_validation(self, canonical_topics,
                            cluster_embeddings,
                            domain="RBI regulatory"):
        """
        canonical_topics: dict {cluster_id: canonical_dict}
        cluster_embeddings: dict {cluster_id: np.array}
        """
        
        # Step 1
        coherence_results = self.run_coherence_validation(
            canonical_topics
        )
        
        # Step 2
        cross_cluster_results = self.find_similar_cluster_pairs(
            canonical_topics, cluster_embeddings
        )
        
        # Step 3
        batch_audit = self.run_batch_audit(
            canonical_topics, domain
        )
        
        # Step 4
        action_plan = self.generate_action_plan(
            coherence_results, cross_cluster_results, batch_audit
        )
        
        return {
            'coherence_results': coherence_results,
            'cross_cluster_results': cross_cluster_results,
            'batch_audit': batch_audit,
            'action_plan': action_plan
        }

validator = ClusterValidator(
    model_endpoint=your_endpoint,
    similarity_threshold=0.75,
    delay=0.3
)

validation_output = validator.run_full_validation(
    canonical_topics=canonical_topics,      # from Stage 3A
    cluster_embeddings=cluster_embeddings,  # dict of np.arrays
    domain="RBI regulatory"
)

action_plan = validation_output['action_plan']

What the Output Tells You

🔴 SPLIT     → re-run topic extraction on that cluster's topics
               with a stricter prompt to force separation

🟡 NOISE     → remove listed topics, reassign to nearest other cluster

🔗 MERGE     → combine two clusters, use proposed canonical name

🟠 GENERIC   → either split into specific subtopics or 
               absorb into neighboring clusters

📋 MISSING   → manually create a topic entry if genuinely absent
               from your 20 documents, or note as a coverage gap
