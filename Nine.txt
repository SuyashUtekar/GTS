import anthropic
import json
import time

client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)

SYSTEM_PROMPT = """You are a senior Indian legal expert and NLP researcher evaluating the quality of topic clusters generated by a BERTopic model on Indian legal documents.

Your job is to evaluate each topic cluster exactly like a human legal expert would — critically and honestly.

You will be given:
- The topic's top keywords (generated by BERTopic)
- ALL text chunks assigned to that topic (complete, no truncation)

You must return ONLY a valid JSON object with this exact structure, no extra text:
{
  "legal_relevance": <int 1-5>,
  "legal_relevance_reason": "<one sentence>",
  "cluster_coherence": <int 1-5>,
  "cluster_coherence_reason": "<one sentence>",
  "keyword_alignment": <int 1-5>,
  "keyword_alignment_reason": "<one sentence>",
  "label_quality": <int 1-5>,
  "suggested_label": "<short human-readable topic name>",
  "label_quality_reason": "<one sentence>",
  "overall_comment": "<2-3 sentence expert summary of this cluster>"
}

Scoring rubric:
- legal_relevance    : 1=no legal content, 3=some legal content, 5=clearly Indian legal domain
- cluster_coherence  : 1=chunks are unrelated, 3=loosely related, 5=all chunks same legal theme
- keyword_alignment  : 1=keywords don't match chunks, 3=partial match, 5=keywords perfectly summarize chunks
- label_quality      : 1=topic label is useless, 3=acceptable, 5=label is precise and informative"""


def build_user_prompt(topic_id, keywords, chunks):
    # ✅ Send ALL chunks, full text, no truncation
    chunks_text = ""
    for i, chunk in enumerate(chunks):
        chunks_text += f"\n[Chunk {i+1}]\n{chunk}\n"

    return f"""Evaluate this topic cluster:

TOPIC ID: {topic_id}
TOP KEYWORDS: {' | '.join(keywords)}
TOTAL CHUNKS IN THIS TOPIC: {len(chunks)}

ALL CHUNKS:
{chunks_text}

Return only the JSON evaluation object."""


def evaluate_topic_with_llm(topic_id, keywords, chunks, retries=3):
    prompt = build_user_prompt(topic_id, keywords, chunks)

    for attempt in range(retries):
        try:
            response = client.messages.create(
                model="claude-sonnet-4-20250514",
                max_tokens=1000,
                system=SYSTEM_PROMPT,
                messages=[{"role": "user", "content": prompt}]
            )
            raw = response.content[0].text.strip()
            raw = raw.replace("```json", "").replace("```", "").strip()
            result = json.loads(raw)
            result["topic_id"]    = topic_id
            result["total_chunks"] = len(chunks)   # ✅ log how many chunks were sent
            return result

        except json.JSONDecodeError as e:
            print(f"   ⚠️  JSON parse error on topic {topic_id}, attempt {attempt+1}: {e}")
            time.sleep(2)
        except Exception as e:
            print(f"   ⚠️  API error on topic {topic_id}, attempt {attempt+1}: {e}")
            time.sleep(3)

    return {
        "topic_id": topic_id,
        "total_chunks": len(chunks),
        "legal_relevance": None, "legal_relevance_reason": "Evaluation failed",
        "cluster_coherence": None, "cluster_coherence_reason": "Evaluation failed",
        "keyword_alignment": None, "keyword_alignment_reason": "Evaluation failed",
        "label_quality": None, "suggested_label": "N/A", "label_quality_reason": "Evaluation failed",
        "overall_comment": "LLM evaluation failed after retries"
    }

print("✅ LLM evaluator ready — sending full chunks, no truncation")
