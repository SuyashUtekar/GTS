def abstractive_single_pass(
    extractive_text,
    summarizer,
    tokenizer,
    min_length = 60,
    max_length = 180,
) -> dict:
    """
    Single pass abstractive summarization.
    Works when extractive summary fits in model context.
    """
    n_tokens = tokenizer(
        extractive_text,
        return_tensors = 'pt',
        truncation     = False
    )['input_ids'].shape[1]

    if n_tokens > tokenizer.model_max_length:
        print(f"  ✗ Too long ({n_tokens} tokens). Use chunked strategy.")
        return None

    result = summarizer(
        extractive_text,
        min_length          = min_length,
        max_length          = max_length,
        num_beams           = 4,
        length_penalty      = 2.0,   # encourages longer output
        early_stopping      = True,
        no_repeat_ngram_size= 3,     # prevents repetition
    )

    return {
        "strategy"          : "single_pass",
        "abstractive_summary": result[0]['summary_text'],
        "input_tokens"      : n_tokens,
    }


Strategy B — Topic-Chunked Pass (if tokens > 1024)

def abstractive_chunked_pass(
    final_selection_ordered,
    sentences,
    partition,
    communities,
    topic_importance,
    summarizer,
    tokenizer,
    max_chunk_tokens = 900,
    min_length       = 30,
    max_length       = 80,
) -> dict:
    """
    Summarize per topic community chunk separately,
    then merge all topic summaries into final abstract.

    Flow:
      Extractive sentences grouped by community
           ↓
      BART summarizes each community chunk
           ↓
      All community summaries merged
           ↓
      Final BART pass on merged text
    """

    # ── Group selected sentences by community ─────────────────
    comm_sentences = defaultdict(list)
    for node in final_selection_ordered:
        comm_id = partition.get(node, -1)
        comm_sentences[comm_id].append(
            sentences[node]['text']
        )

    # sort communities by topic importance (most → least)
    sorted_comms = sorted(
        comm_sentences.keys(),
        key     = lambda c: topic_importance.get(c, 0),
        reverse = True
    )

    print(f"Processing {len(sorted_comms)} topic chunks...")
    print("-" * 55)

    # ── Summarize each community chunk ────────────────────────
    community_summaries = {}

    for comm_id in sorted_comms:
        chunk_text  = " ".join(comm_sentences[comm_id])
        n_tokens    = tokenizer(
            chunk_text,
            return_tensors = 'pt',
            truncation     = False
        )['input_ids'].shape[1]

        ti   = topic_importance.get(comm_id, 0)
        print(f"\n  Community {comm_id}  "
              f"(importance={ti:.3f}, "
              f"sentences={len(comm_sentences[comm_id])}, "
              f"tokens={n_tokens})")

        if n_tokens < 30:
            # too short to summarize → keep as is
            community_summaries[comm_id] = chunk_text
            print(f"  → Too short, kept as is")
            continue

        # truncate if exceeds model max
        if n_tokens > max_chunk_tokens:
            chunk_text = tokenizer.decode(
                tokenizer(
                    chunk_text,
                    max_length = max_chunk_tokens,
                    truncation = True
                )['input_ids'],
                skip_special_tokens = True
            )
            print(f"  → Truncated to {max_chunk_tokens} tokens")

        # scale output length by topic importance
        chunk_max_len = max(
            40,
            int(max_length * ti * len(sorted_comms))
        )
        chunk_max_len = min(chunk_max_len, max_length)

        result = summarizer(
            chunk_text,
            min_length           = min_length,
            max_length           = chunk_max_len,
            num_beams            = 4,
            length_penalty       = 1.5,
            early_stopping       = True,
            no_repeat_ngram_size = 3,
        )

        community_summaries[comm_id] = result[0]['summary_text']
        print(f"  → \"{result[0]['summary_text'][:80]}...\"")

    # ── Merge all community summaries ─────────────────────────
    merged = " ".join([
        community_summaries[cid]
        for cid in sorted_comms
        if cid in community_summaries
    ])

    print(f"\nMerging {len(community_summaries)} topic summaries...")
    print(f"  Merged text tokens: "
          f"{tokenizer(merged, return_tensors='pt', truncation=False)['input_ids'].shape[1]}")

    # ── Final fusion pass ─────────────────────────────────────
    merged_tokens = tokenizer(
        merged,
        return_tensors = 'pt',
        truncation     = False
    )['input_ids'].shape[1]

    if merged_tokens > tokenizer.model_max_length:
        merged = tokenizer.decode(
            tokenizer(
                merged,
                max_length = tokenizer.model_max_length,
                truncation = True
            )['input_ids'],
            skip_special_tokens = True
        )

    final_result = summarizer(
        merged,
        min_length           = 80,
        max_length           = 220,
        num_beams            = 4,
        length_penalty       = 2.0,
        early_stopping       = True,
        no_repeat_ngram_size = 3,
    )

    return {
        "strategy"            : "topic_chunked",
        "community_summaries" : community_summaries,
        "merged_text"         : merged,
        "abstractive_summary" : final_result[0]['summary_text'],
        "n_chunks"            : len(sorted_comms),
    }

# ── Fluency proxy: avg sentence length variance ───────────────
def fluency_score(text):
    """
    Proxy for fluency:
    Lower variance in sentence length = more fluent.
    Extractive summaries have high variance
    (sentences from different parts of doc).
    Abstractive summaries are more uniform.
    """
    sents  = text.split('.')
    sents  = [s.strip() for s in sents if len(s.strip()) > 0]
    lens   = [len(s.split()) for s in sents]
    if len(lens) < 2:
        return 0.0
    return round(1.0 / (1.0 + np.std(lens)), 4)

# ── Novelty: words in abstractive NOT in extractive ───────────
def novelty_score(abstractive, extractive):
    """
    What fraction of abstractive words are NEW
    (not copied from extractive summary).
    Higher = more abstractive / generative.
    Lower = mostly copying extractive phrases.
    """
    ext_words = set(extractive.lower().split())
    abs_words = abstractive.lower().split()
    novel     = [w for w in abs_words if w not in ext_words]
    return round(len(novel) / len(abs_words), 4) if abs_words else 0.0
